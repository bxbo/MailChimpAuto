create table emails (
	email varchar(100),
	ind_id int,
	email_id int
);

create table emails_link (
	email_id int,
	mailchimp_id varchar(100)
);

create table mailchimp_campaigns (
	id varchar(100),
	web_id varchar(100),
    status varchar(100),
	emails_sent int,
	send_time timestamp,
	unique_opens int,
	open_rate int,
	clicks int,
	click_rate int,
	last_sync_date timestamp
);

create table mailchimp_audiences (
	id varchar(100),
	web_id varchar(100),
    name varchar(100),
    date_created timestamp,
    member_count int,
    unsubscribe_count int,
	last_sub_date timestamp,
	last_unsub_date timestamp
);
